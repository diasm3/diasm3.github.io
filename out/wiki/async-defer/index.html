<!DOCTYPE html>
<html lang="en">

        <meta charset="UTF-8" />

        //미리 3개다 읽고 순서대로 불러온다
        // a -> b -> c
        
        <script defer src="a.js"></script>
        <script defer src="b.js"></script>
        <script defer src="c.js"></script>
        
    </head>

    <div></div>
