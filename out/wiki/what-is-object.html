Fun cloning

1const user = {name: 'ellie', age: '20'}
2
3const user2 = user
4user2.name = 'coder'
5console.log(user) // 두개다 객체의 주소를 가르키기 때문에 가르키고 있는 내용은 변경된다
6
7//따라서 오브젝트를 그대로 복사하고 싶으면
8const user3 = {}
9for (key in user){
10    user3[key] = user[key]
11}
12console.log(user3) //옛날 방법
13
14//다른 방법으로는
15//기본지원으로 언제든지 사용할 수 있음
